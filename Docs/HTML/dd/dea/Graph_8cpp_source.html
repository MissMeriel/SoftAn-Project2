84 void Graph::addBranchEdges() {
85  for (auto func : graphFunctions) {
86  std::list<Graph_Line *> lines = func->getFunctionLines();
87  for (auto line = lines.begin(); line != lines.end(); ++line) {
88  std::list<Graph_Instruction *> instructions = (*line)->getLineInstructions();
89  for (auto inst = instructions.begin(); inst != instructions.end(); ++inst) {
90  llvm::Instruction *I = (*inst)->getInstructionPtr();
91  if (I) {
92  /* Adding edges for BB with multiple successors */
93  if (I->isTerminator()) {
94  unsigned int noSucc = I->getNumSuccessors();
95  for (unsigned int iterSucc = 0; iterSucc < noSucc; ++iterSucc) {
96  llvm::Instruction *iSucc = llvm::dyn_cast<llvm::Instruction>(I->getSuccessor(iterSucc)->begin());
97  Graph_Instruction *iSuccInst = findMatchedInstruction(iSucc);
98  if (iSucc) {
99  Graph_Edge *branchEdge = new Graph_Edge(*inst, iSuccInst, Graph_Edge::BRANCH, graphVersion);
100  addEdge(*inst, iSuccInst, branchEdge);
101  } else {
102  std::cerr << "No matching Graph_Instruction found for edge from " << (*inst)->getInstructionLabel()
103  << "\n";
104  } // End check for iSucc
105  } // End loop for iterSucc
106  } else if ((*inst)->getInstructionID() == instructions.back()->getInstructionID()) {
107  /* Adding Unique successors */
108  auto nextLine = std::next(line);
109  if (nextLine != lines.end()) {
110  std::list<Graph_Instruction *> nextInstructions = (*nextLine)->getLineInstructions();

112  if (nextI != nextInstructions.end()) {
113  Graph_Edge *seqEdge = new Graph_Edge(*inst, *nextI, Graph_Edge::SEQUENTIAL, graphVersion);
114  addEdge(*inst, *nextI, seqEdge);
115  } // End check for nextI

117  } // End check for TerminatorInst
118  } // End check for I

125  /* External Node */
126  Graph_Function *virtualNodeFunc = new Graph_Function(getNextID());
127  virtualNodeFunc->setFunctionFile("External_Node_File");
128  virtualNodeFunc->setFunctionName("External_Node_Func");
129  Graph_Line *virtualNodeLine = new Graph_Line(graphVersion);
130  virtualNodeLine->setLineNumber(graphVersion, graphEntryID);
131  Graph_Instruction *externalNode = new Graph_Instruction();
132  externalNode->setInstructionID(getNextID());
133  externalNode->setInstructionLabel("External_Node");
134  externalNode->setInstructionPtr(NULL);
135  virtualNodeLine->pushLineInstruction(externalNode);
136  virtualNodeFunc->pushFunctionLines(virtualNodeLine);
137  pushGraphFunction(virtualNodeFunc);
138  std::list<std::string> funcNotFoud;
139  /* Get the whitelisted function names and merge with funcNotFoud */
140  std::list<std::string> whiteListedFunc = Graph::getWhiteList();
141  funcNotFoud.insert(funcNotFoud.end(), whiteListedFunc.begin(), whiteListedFunc.end());
142  for (auto func : graphFunctions) {
143  for (auto line : func->getFunctionLines()) {
144  for (auto inst : line->getLineInstructions()) {
145  llvm::Instruction *I = inst->getInstructionPtr();
146  if (I) {
147  if (auto callSite = llvm::CallSite(I)) {
148  const llvm::Function *Callee = callSite.getCalledFunction();
149  if (!Callee || !llvm::Intrinsic::isLeaf(Callee->getIntrinsicID())) {
150  /* Call Extern */
151  Graph_Edge *callEdge = new Graph_Edge(inst, externalNode, Graph_Edge::EXTERNAL_CALL, graphVersion);
152  addEdge(inst, externalNode, callEdge);
153  } else if (!Callee->isIntrinsic()) {
154  /* Add Edge based on function name */
155  bool noEntry = false;
156  bool noExit = false;
157  if (!Callee->getName().empty()) {
158  std::string funcName = Callee->getName();
159  /* Call site to Entry */
160  Graph_Instruction *virtualEntry = findVirtualEntry(funcName);
161  if (virtualEntry) {
162  Graph_Edge *callEdge = new Graph_Edge(inst, virtualEntry, Graph_Edge::CALL, graphVersion);
163  addEdge(inst, virtualEntry, callEdge);
164  } else {
165  noEntry = true;
166  } // End check for virtualEntry
167  /* Exit to Call site */
168  Graph_Instruction *virtualExit = findVirtualExit(funcName);
169  if (virtualExit) {
170  Graph_Edge *callEdge = new Graph_Edge(virtualExit, inst, Graph_Edge::CALL, graphVersion);
171  addEdge(virtualExit, inst, callEdge);
172  } else {
173  noExit = true;
174  } // End check for virtualExit
175  auto findFunc = std::find_if(std::begin(funcNotFoud), std::end(funcNotFoud),
176  [=](std::string name) { return name == funcName; });
177  if (findFunc == funcNotFoud.end()) {
178  if (noEntry && noExit) {
179  funcNotFoud.push_back(funcName);
180  std::cerr << "Call edges not formed for " << funcName << "\n";
181  } else if (noEntry) {
182  std::cerr << "No Virtual Entry found for " << funcName << "\n";
183  } else if (noExit) {
184  std::cerr << "No Virtual Exit found for " << funcName << "\n";
185  } // End check for noEntry & noExit combinations
186  } // End check for findFunc
187  } else {
188  std::cerr << "Unknown function call from Instruction " << inst->getInstructionLabel() << "\n";
189  } // End check for Callee name
190  } // End check for Callee Intrinsic
191  } // End check for callSite
192  } // End check for I
193  } // End loop for inst
194  } // End loop for line
195  } // End loop for func
196 } // End addFunctionCallEdges

296 void getLocationInfo(llvm::Instruction &I, unsigned int &DILocLine, std::string &DIFile) {
297  if (llvm::DILocation *DILoc = I.getDebugLoc()) {
298  DILocLine = DILoc->getLine();

300  return;
301  } else {
302  bool resolvedLine = false;
303  /* Search backward */
304  llvm::Instruction *I_iter = &I;
305  do {
306  auto prev = I_iter->getPrevNode();
307  if (prev) {

309  DILocLine = prev->getDebugLoc()->getLine();
310  DIFile = prev->getDebugLoc()->getFilename();
311  resolvedLine = true;
312  return;
313  } // End check for getDebugLoc

315  } else {
316  break;
317  } // End check for prev

319  /* Search forward */
320  if (!resolvedLine) {
321  llvm::Instruction *I_iter = &I;
322  do {
323  auto next = I_iter->getNextNode();
324  if (next) {

327  DIFile = next->getDebugLoc()->getFilename();

330  } // End check for getDebugLoc
331  I_iter = I_iter->getNextNode();
332  } else {
